\section{General Solution for Repeating Roots}
Let one root, $y_1$, be found as
\begin{equation}
    y_1 = e^{-\frac{b}{2a}x}
\end{equation}
The other root, $y_2$ must not be a constant multiple of $y_1$. Thus, it can be expressed as follows: 
\begin{equation}
    y_2 = v(x)e^{-\frac{b}{2a}x}
\end{equation}
What continues is to find $v(x)$ that defines $y_2$
\begin{equation}
    y_2' = v'(x)e^{-\frac{b}{2a}x}
	   -\frac{b}{2a} v(x)e^{-\frac{b}{2a}x}
\end{equation}
\begin{equation}
    y_2'' =
        v''(x)e^{-\frac{b}{2a}x}
        -\frac{b}{2a}v'(x)e^{-\frac{b}{2a}x}
        -\frac{b}{2a}v'(x)e^{-\frac{b}{2a}x}
        +\frac{b^2}{4a^2}v(x)e^{-\frac{b}{2a}x}
\end{equation}
Substitute into the original equation: 
$$
ay'' + by' + cy = 0
$$

\begin{equation}
\begin{split}
    av''(x)e^{-\frac{b}{2a}x}
    &- a\frac{b}{2a}v'(x)e^{-\frac{b}{2a}x}
    - a\frac{b}{2a}v'(x)e^{-\frac{b}{2a}x} \\
    &+ a\frac{b^2}{4a^2}v(x)e^{-\frac{b}{2a}x}
    + bv'(x)e^{-\frac{b}{2a}x} \\
    &- b\frac{b}{2a} v(x)e^{-\frac{b}{2a}x}
    + c v(x)e^{-\frac{b}{2a}x}
    = 0
\end{split}
\end{equation}
Simplify:
\[
\begin{split}
    av''(x)e^{-\frac{b}{2a}x}
    &-\frac{b}{2}v'(x)e^{-\frac{b}{2a}x}
    -\frac{b}{2}v'(x)e^{-\frac{b}{2a}x}\\
    &+\frac{b^2}{4a}v(x)e^{-\frac{b}{2a}x}
    +bv'(x)e^{-\frac{b}{2a}x}\\
    &-\frac{b^2}{2a} v(x)e^{-\frac{b}{2a}x}
    + cv(x)e^{-\frac{b}{2a}x}
    = 0
\end{split}
\]
Combine like terms:
\[
    av''(x)e^{-\frac{b}{2a}x}
    -\frac{b^2}{4a}v(x)e^{-\frac{b}{2a}x}
    + cv(x)e^{-\frac{b}{2a}x}
    = 0
\]
Factor:
\begin{equation}
    e^{-\frac{b}{2a}x}
    \left[
    a v''(x)
    + \left(c - \frac{b^2}{4a}\right)v(x)
    \right]
    = 0
\end{equation}
Since $e^{-\frac{b}{2a}x}\neq 0$:
\begin{equation}
    a v''(x)
    = \left(\frac{b^2}{4a} - c\right)v(x)
\end{equation}
Multiply by $4a$:
$$
4a^2 v''(x) = (b^2 - 4ac)\, v(x)
$$
For repeated roots, $b^2 - 4ac = 0$:
$$
4a^2 v''(x) = 0
$$
$$
v''(x) = 0
$$
$$
v'(x) = p
$$
\begin{equation}
    v(x) = px + q
\end{equation}
Final form:
\begin{equation}
y_2 = (px + q)e^{-\frac{b}{2a}x}
\end{equation}
The general solution of the differential equation is: 
$$
y = c_1y_1+c_2y_2 
$$
$$y=c_1e^{-\frac{b}{2a}x} + c_2(px+q)e^{-\frac{b}{2a}x}$$
This could be rearranged and simplified into: 
\begin{equation}
    y = (Ax+B)e^{-\frac{b}{2a}x}
\end{equation}

\section{General Solution for Complex Roots}

Roots of auxiliary equation: $m=p\pm qi$
\begin{equation}
\begin{split}
y_1=e^{px}e^{ipx}=e^{px}(\cos(qx)+i\sin(qx)) \\
y_2=e^{px}e^{-ipx}=e^{px}(\cos(qx)-i\sin(qx)) \\
\end{split}
\end{equation}
All linear combinations of the roots $y_1$ and $y_2$ are solutions: 
\begin{equation}
    u(x)=\frac12 (y_1+y_2)=e^{px}(\cos(qx))
\end{equation}

\begin{equation}
    v(x)=\frac{1}{2i}(y_1-y_2)=e^{px}\sin(qx)
\end{equation}
$$y=Au(x)+Bv(x)$$
\begin{equation}
    y=e^{px}\left( A\cos(qx)+B\sin(qx) \right)
\end{equation}

\section{Particular Integral for Non-Homogeneous Equations}

Suppose $Y_1$ and $Y_2$ are solutions to a differential equation
$$ay''+by'+cy=f(x)$$
For all solutions, the following holds true: 
$$Y_1-Y_2=Ay_1+By_2$$
where $y_1$ and $y_2$ are a fundamental set of solutions to the differential equation. 
\
Now suppose $y$ is the general solution and $Y_P$ is any solution known to us: 
$$y-Y_P = Ay_1+By_2$$
\begin{equation}
    y = Ay_1+By_2 + Y_P
\end{equation}
Note that $Ay_1+By_2$ is the complementary function and $Y_P$ is the particular integral.
